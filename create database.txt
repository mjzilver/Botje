CREATE TABLE `images` (
	`link`	TEXT NOT NULL UNIQUE,
	`sub`	TEXT NOT NULL,
	PRIMARY KEY(`link`)
);

CREATE TABLE `idea` (
	`id`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`user`	TEXT NOT NULL,
	`idea`	TEXT NOT NULL
);

CREATE TABLE "farm" (
	"user_id"	INTEGER NOT NULL UNIQUE,
	"yield"	INTEGER,
	"tier"	INTEGER,
	"time"	INTEGER,
	"points"	INTEGER,
	"planted_at"	INTEGER,
	PRIMARY KEY("user_id")
);
